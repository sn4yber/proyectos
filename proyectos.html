Este es un proyecto de CRUD en Java

Este es un ejemplo de código Java que es un sistema de cursos de universidad. Se conecta a una base de datos PostgreSQL y realiza operaciones CRUD (Crear, Leer, Actualizar, Eliminar) en una tabla llamada "cursos".


package org.example.sna;

import java.sql.*;
import java.util.Scanner;

public class App {
    public static void main(String[] args) {
        String URL = "jdbc:postgresql://localhost:5432/postgres";
        String USER = "postgres";
        String PASSWORD = "456789";

        Scanner scanner = new Scanner(System.in);

        try (Connection conn = DriverManager.getConnection(URL, USER, PASSWORD)) {
            System.out.println("Conectado a la base de datos.");
        } catch (SQLException e) {
            System.out.println("Error al conectar: " + e.getMessage());
        }
    }
}
int option;
            do {
                System.out.println("\n--- Menú ---");
                System.out.println("1. Listar todos los cursos");
                System.out.println("2. Buscar curso por ID");
                System.out.println("3. Actualizar curso");
                System.out.println("4. Eliminar curso");
                System.out.println("5. Insertar curso");
                System.out.println("6. Salir");
                System.out.print("Seleccione una opción: ");

                option = scanner.nextInt();
                scanner.nextLine();

                switch (option) {
                    case 1:
                        System.out.println("Listando cursos...");

                        // Consulta SQL para obtener todos los cursos
                        String query = "SELECT * FROM cursos";  // Consulta básica
                        try (PreparedStatement stmt = conn.prepareStatement(query);
                             ResultSet rs = stmt.executeQuery()) {

                            // Verificamos si el ResultSet tiene resultados
                            boolean hayCursos = false;
                            while (rs.next()) {
                                int id = rs.getInt("id");
                                String nombre = rs.getString("nombre");
                                int creditos = rs.getInt("creditos");
                                int idProfesor = rs.getInt("id_profesor");

                                // Mostrar cada curso en consola
                                System.out.println("ID: " + id + ", Nombre: " + nombre +
                                        ", Créditos: " + creditos + ", ID Profesor: " + idProfesor);
                                hayCursos = true;
                            }

                            // Si no hay cursos, mostrar un mensaje
                            if (!hayCursos) {
                                System.out.println("No se encontraron cursos.");
                            }

                        } catch (SQLException e) {
                            System.err.println("Error al listar cursos: " + e.getMessage());
                        }
                        break;


                    case 2:
                        System.out.print("Ingrese el ID del curso: ");
                        int idBuscar = scanner.nextInt();
                        scanner.nextLine();
                        try (PreparedStatement stmt = conn.prepareStatement("SELECT * FROM cursos WHERE id = ?")) {
                            stmt.setInt(1, idBuscar);
                            try (ResultSet rs = stmt.executeQuery()) {
                                if (rs.next()) {
                                    System.out.println("ID: " + rs.getInt("id") +
                                            ", Nombre: " + rs.getString("nombre") +
                                            ", Créditos: " + rs.getInt("creditos") +
                                            ", ID Profesor: " + rs.getInt("id_profesor"));
                                } else {
                                    System.out.println("Curso no encontrado.");
                                }
                            }
                        } catch (SQLException e) {
                            System.err.println("Error al buscar curso: " + e.getMessage());
                        }
                        break;

                    case 3:
                        System.out.print("ID del curso a actualizar: ");
                        int idActualizar = scanner.nextInt();
                        scanner.nextLine();
                        System.out.print("Nuevo nombre: ");
                        String nuevoNombre = scanner.nextLine();
                        System.out.print("Nuevos créditos: ");
                        int nuevosCreditos = scanner.nextInt();
                        System.out.print("Nuevo ID del profesor: ");
                        int nuevoProfesor = scanner.nextInt();
                        scanner.nextLine();

                        try (PreparedStatement stmt = conn.prepareStatement(
                                "UPDATE cursos SET nombre = ?, creditos = ?, id_profesor = ? WHERE id = ?")) {
                            stmt.setString(1, nuevoNombre);
                            stmt.setInt(2, nuevosCreditos);
                            stmt.setInt(3, nuevoProfesor);
                            stmt.setInt(4, idActualizar);

                            int filas = stmt.executeUpdate();
                            System.out.println("Curso actualizado. Filas afectadas: " + filas);
                        } catch (SQLException e) {
                            System.err.println("Error al actualizar curso: " + e.getMessage());
                        }
                        break;

                    case 4:
                        System.out.print("ID del curso a eliminar: ");
                        int idEliminar = scanner.nextInt();
                        scanner.nextLine();
                        try (PreparedStatement stmt = conn.prepareStatement("DELETE FROM cursos WHERE id = ?")) {
                            stmt.setInt(1, idEliminar);
                            int filas = stmt.executeUpdate();
                            System.out.println("Curso eliminado. Filas afectadas: " + filas);
                        } catch (SQLException e) {
                            System.err.println("Error al eliminar curso: " + e.getMessage());
                        }
                        break;

                    case 5:
                        System.out.print("Nombre del curso: ");
                        String nombre = scanner.nextLine();
                        System.out.print("Créditos: ");
                        int creditos = scanner.nextInt();
                        System.out.print("ID del profesor: ");
                        int idProfesor = scanner.nextInt();
                        scanner.nextLine();

                        try (PreparedStatement stmt = conn.prepareStatement(
                                "INSERT INTO cursos (nombre, creditos, id_profesor) VALUES (?, ?, ?)")) {
                            stmt.setString(1, nombre);
                            stmt.setInt(2, creditos);
                            stmt.setInt(3, idProfesor);

                            int filas = stmt.executeUpdate();
                            System.out.println("Curso insertado. Filas afectadas: " + filas);
                        } catch (SQLException e) {
                            System.err.println("Error al insertar curso: " + e.getMessage());
                        }
                        break;

                    case 6:
                        System.out.println("Saliendo del sistema...");
                        break;

                    default:
                        System.out.println("Opción no válida.");
                }

            } while (option != 6);

        } catch (SQLException e) {
            System.err.println("Error al conectar con la base de datos: " + e.getMessage());
        }

        scanner.close();
    }
}
